```python
from flask import Flask, request, jsonify, render_template
from flask_sqlalchemy import SQLAlchemy
from werkzeug.security import generate_password_hash, check_password_hash
import requests
import os
from dotenv import load_dotenv
import jwt
import datetime
from functools import wraps
import stripe
import paypalrestsdk

# تحميل المتغيرات البيئية
load_dotenv()

app = Flask(__name__)

# إعداد قاعدة البيانات
app.config['SQLALCHEMY_DATABASE_URI'] = 'postgresql://user:password@localhost/payment_db'
app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False
app.config['SECRET_KEY'] = os.getenv('SECRET_KEY')
app.config['STRIPE_SECRET_KEY'] = os.getenv('STRIPE_SECRET_KEY')
app.config['PAYPAL_CLIENT_ID'] = os.getenv('PAYPAL_CLIENT_ID')
app.config['PAYPAL_CLIENT_SECRET'] = os.getenv('PAYPAL_CLIENT_SECRET')

db = SQLAlchemy(app)

# إعداد Stripe
stripe.api_key = app.config['STRIPE_SECRET_KEY']

# إعداد PayPal
paypalrestsdk.configure({
    "mode": "sandbox",  # أو "live" للإنتاج
    "client_id": app.config['PAYPAL_CLIENT_ID'],
    "client_secret": app.config['PAYPAL_CLIENT_SECRET']
})

# نماذج قاعدة البيانات
class User(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    username = db.Column(db.String(80), unique=True, nullable=False)
    email = db.Column(db.String(120), unique=True, nullable=False)
    password_hash = db.Column(db.String(128))
    phone = db.Column(db.String(20))
    two_factor_secret = db.Column(db.String(32))

    def set_password(self, password):
        self.password_hash = generate_password_hash(password)

    def check_password(self, password):
        return check_password_hash(self.password_hash, password)

class PaymentMethod(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'), nullable=False)
    type = db.Column(db.String(20), nullable=False)  # 'credit_card' or 'paypal'
    token = db.Column(db.String(255), nullable=False)  # رمز مميز للبطاقة أو PayPal
    last_four = db.Column(db.String(4))  # آخر 4 أرقام للبطاقة
    is_default = db.Column(db.Boolean, default=False)

class Transaction(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'), nullable=False)
    amount = db.Column(db.Float, nullable=False)
    currency = db.Column(db.String(3), default='USD')
    payment_method_id = db.Column(db.Integer, db.ForeignKey('payment_method.id'))
    status = db.Column(db.String(20), default='pending')  # pending, completed, failed
    transaction_id = db.Column(db.String(255))  # معرف المعاملة من مزود الدفع
    created_at = db.Column(db.DateTime, default=datetime.datetime.utcnow)

# دالة للتحقق من التوكن
def token_required(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        token = request.headers.get('Authorization')
        if not token:
            return jsonify({'message': 'Token is missing!'}), 403
        try:
            data = jwt.decode(token, app.config['SECRET_KEY'], algorithms=["HS256"])
            current_user = User.query.get(data['user_id'])
        except:
            return jsonify({'message': 'Token is invalid!'}), 403
        return f(current_user, *args, **kwargs)
    return decorated

# مسارات API
@app.route('/register', methods=['POST'])
def register():
    data = request.get_json()

    # التحقق من البيانات
    if not data or not data.get('username') or not data.get('email') or not data.get('password'):
        return jsonify({'message': 'Missing required fields'}), 400

    # التحقق من وجود المستخدم
    if User.query.filter_by(username=data['username']).first():
        return jsonify({'message': 'Username already exists'}), 400

    if User.query.filter_by(email=data['email']).first():
        return jsonify({'message': 'Email already exists'}), 400

    # إنشاء المستخدم
    user = User(username=data['username'], email=data['email'])
    user.set_password(data['password'])
    db.session.add(user)
    db.session.commit()

    return jsonify({'message': 'User created successfully'}), 201

@app.route('/login', methods=['POST'])
def login():
    data = request.get_json()

    if not data or not data.get('username') or not data.get('password'):
        return jsonify({'message': 'Missing username or password'}), 400

    user = User.query.filter_by(username=data['username']).first()

    if not user or not user.check_password(data['password']):
        return jsonify({'message': 'Invalid username or password'}), 401

    # إنشاء توكن JWT
    token = jwt.encode({
        'user_id': user.id,
        'exp': datetime.datetime.utcnow() + datetime.timedelta(hours=24)
    }, app.config['SECRET_KEY'], algorithm="HS256")

    return jsonify({'token': token}), 200

@app.route('/add_payment_method', methods=['POST'])
@token_required
def add_payment_method(current_user):
    data = request.get_json()

    if not data or not data.get('type') or not data.get('token'):
        return jsonify({'message': 'Missing required fields'}), 400

    # التحقق من نوع طريقة الدفع
    if data['type'] not in ['credit_card', 'paypal']:

    # إنشاء طريقة دفع جديدة
    payment_method = PaymentMethod(
        user_id=current_user.id,
        type=data['type'],
        token=data['token'],
        last_four=data.get('last_four', ''),
        is_default=data.get('is_default', False)
    )

    db.session.add(payment_method)
    db.session.commit()

    return jsonify({'message': 'Payment method added successfully'}), 201

@app.route('/create_payment', methods=['POST'])
@token_required
def create_payment(current_user):
    data = request.get_json()

    if not data or not data.get('amount') or not data.get('payment_method_id'):
        return jsonify({'message': 'Missing required fields'}), 400

    # التحقق من طريقة الدفع
    payment_method = PaymentMethod.query.filter_by(
        id=data['payment_method_id'],
        user_id=current_user.id
    ).first()

    if not payment_method:
        return jsonify({'message': 'Payment method not found'}), 404

    # إنشاء معاملة جديدة
    transaction = Transaction(
        user_id=current_user.id,
        amount=data['amount'],
        payment_method_id=payment_method.id,
        currency=data.get('currency', 'USD')
    )

    db.session.add(transaction)
    db.session.commit()

    try:
        if payment_method.type == 'credit_card':
            # معالجة الدفع ببطاقة الائتمان باستخدام Stripe
            charge = stripe.Charge.create(
                amount=int(data['amount'] * 100),  # المبلغ بالقرش
                currency=data.get('currency', 'USD'),
                source=payment_method.token,
                description=f'Payment for user {current_user.id}'
            )

            transaction.status = 'completed'
            transaction.transaction_id = charge.id

        elif payment_method.type == 'paypal':
            # معالجة الدفع باستخدام PayPal
            payment = paypalrestsdk.Payment({
                "intent": "sale",
                "payer": {
                    "payment_method": "paypal"
                },
                "redirect_urls": {
                    "return_url": "http://localhost:5000/payment/execute",
                    "cancel_url": "http://localhost:5000/payment/cancel"
                },
                "transactions": [{
                    "item_list": {
                        "items": [{
                            "name": "Payment",
                            "sku": "item",
                            "price": str(data['amount']),
                            "currency": data.get('currency', 'USD'),
                            "quantity": 1
                        }]
                    },
                    "amount": {
                        "total": str(data['amount']),
                        "currency": data.get('currency', 'USD')
                    },
                    "description": f"Payment for user {current_user.id}"
                }]
            })

            if payment.create():
                # تخزين معرف الدفع مؤقتًا
                transaction.transaction_id = payment.id
                db.session.commit()

                # إعادة توجيه المستخدم إلى PayPal للموافقة
                for link in payment.links:
                    if link.rel == "approval_url":
                        approval_url = link.href
                        return jsonify({
                            'message': 'Payment requires approval',
                            'approval_url': approval_url,
                            'transaction_id': transaction.id
                        }), 200

            else:
                transaction.status = 'failed'
                return jsonify({'message': payment.error}), 400

        db.session.commit()
        return jsonify({
            'message': 'Payment processed successfully',
            'transaction_id': transaction.id,
            'status': transaction.status
        }), 200

    except Exception as e:
        transaction.status = 'failed'
        db.session.commit()
        return jsonify({'message': str(e)}), 400

@app.route('/payment/execute', methods=['GET'])
def execute_payment():
    payment_id = request.args.get('paymentId')
    payer_id = request.args.get('PayerID')

    payment = paypalrestsdk.Payment.find(payment_id)

    if payment.execute({"payer_id": payer_id}):
        # تحديث حالة المعاملة
        transaction = Transaction.query.filter_by(transaction_id=payment_id).first()
        if transaction:
            transaction.status = 'completed'
            db.session.commit()

        return jsonify({'message': 'Payment executed successfully'}), 200
    else:
        return jsonify({'message': payment.error}), 400

if __name__ == '__main__':
    with app.app_context():
        db.create_all()
    app.run(ssl_context='adhoc', debug=True)
```

## 3. إجراءات الأمان المتقدمة

### أ. حماية البيانات الحساسة

1. **تشفير البيانات في قاعدة البيانات**:
   ```python
   from cryptography.fernet import Fernet

   # توليد مفتاح التشفير (يجب تخزينه بأمان)
   key = Fernet.generate_key()
   cipher_suite = Fernet(key)

   # تشفير البيانات
   def encrypt_data(data):
       return cipher_suite.encrypt(data.encode())

   # فك تشفير البيانات
   def decrypt_data(encrypted_data):
       return cipher_suite.decrypt(encrypted_data).decode()
   ```

2. **استخدام الرموز المميزة (Tokens)**:
   - بدلاً من تخزين أرقام البطاقات الكاملة، نستخدم رموز مميزة من مزودي الدفع مثل Stripe أو PayPal

### ب. التحقق الثنائي العامل (2FA)

```python
import pyotp

# عند تسجيل المستخدم
def setup_2fa(user):
    totp = pyotp.TOTP(pyotp.random_base32())
    user.two_factor_secret = totp.secret
    db.session.commit()
    return totp.provisioning_uri(user.email, issuer_name="SecurePay")

# عند التحقق من 2FA
def verify_2fa(user, token):
    totp = pyotp.TOTP(user.two_factor_secret)
    return totp.verify(token)
```

```python
def token_required(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        token = request.headers.get('Authorization')
        if not token:
            return jsonify({'message': 'Token is missing!'}), 403
        try:
            data = jwt.decode(token, app.config['SECRET_KEY'], algorithms=["HS256"])
            current_user = User.query.get(data['user_id'])
        except:
            return jsonify({'message': 'Token is invalid!'}), 403
        return f(current_user, *args, **kwargs)
    return decorated

@app.route('/register', methods=['POST'])
def register():
    data = request.get_json()

    # التحقق من البيانات
    if not data or not data.get('username') or not data.get('email') or not data.get('password'):
        return jsonify({'message': 'Missing required fields'}), 400

    # التحقق من وجود المستخدم
    if User.query.filter_by(username=data['username']).first():
        return jsonify({'message': 'Username already exists'}), 400

    if User.query.filter_by(email=data['email']).first():
        return jsonify({'message': 'Email already exists'}), 400

    # إنشاء المستخدم
    user = User(username=data['username'], email=data['email'])
    user.set_password(data['password'])
    db.session.add(user)
    db.session.commit()

    return jsonify({'message': 'User created successfully'}), 201

    user = User.query.filter_by(username=data['username']).first()

    if not user or not user.check_password(data['password']):
        return jsonify({'message': 'Invalid username or password'}), 401

    # إنشاء توكن JWT
    token = jwt.encode({
        'user_id': user.id,
        'exp': datetime.datetime.utcnow() + datetime.timedelta(hours=24)
    }, app.config['SECRET_KEY'], algorithm="HS256")

    return jsonify({'token': token}), 200

@app.route('/add_payment_method', methods=['POST'])

def add_payment_method(current_user):
    data = request.get_json()

@app.route('/create_payment', methods=['POST'])
@token_required
def create_payment(current_user):
    data = request.get_json()

    if not data or not data.get('amount') or not data.get('payment_method_id'):
        return jsonify({'message': 'Missing required fields'}), 400

    # التحقق من طريقة الدفع
    payment_method = PaymentMethod.query.filter_by(
        id=data['payment_method_id'],
        user_id=current_user.id
    ).first()

    if not payment_method:
        return jsonify({'message': 'Payment method not found'}), 404

    # إنشاء معاملة جديدة
    transaction = Transaction(
        user_id=current_user.id,
        amount=data['amount'],
        payment_method_id=payment_method.id,
        currency=data.get('currency', 'USD')
    )

    db.session.add(transaction)
    db.session.commit()

    try:
        if payment_method.type == 'credit_card':
            # معالجة الدفع ببطاقة الائتمان باستخدام Stripe
            charge = stripe.Charge.create(
                amount=int(data['amount'] * 100),  # المبلغ بالقرش
                currency=data.get('currency', 'USD'),
                source=payment_method.token,
                description=f'Payment for user {current_user.id}'
            )

            transaction.status = 'completed'
            transaction.transaction_id = charge.id

        elif payment_method.type == 'paypal':
            # معالجة الدفع باستخدام PayPal
            payment = paypalrestsdk.Payment({
                "intent": "sale",
                "payer": {
                    "payment_method": "paypal"
                },
                "redirect_urls": {
                    "return_url": "http://localhost:5000/payment/execute",
                    "cancel_url": "http://localhost:5000/payment/cancel"
                },
                "transactions": [{
                    "item_list": {
                        "items": [{
                            "name": "Payment",
                            "sku": "item",
                            "price": str(data['amount']),
                            "currency": data.get('currency', 'USD'),
                            "quantity": 1
                        }]
                    },
                    "amount": {
                        "total": str(data['amount']),
                        "currency": data.get('currency', 'USD')
                    },
                    "description": f"Payment for user {current_user.id}"
                }]
            })

            if payment.create():
                # تخزين معرف الدفع مؤقتًا
                transaction.transaction_id = payment.id
                db.session.commit()

                # إعادة توجيه المستخدم إلى PayPal للموافقة
                for link in payment.links:
                    if link.rel == "approval_url":
                        approval_url = link.href
                        return jsonify({
                            'message': 'Payment requires approval',
                            'approval_url': approval_url,
                            'transaction_id': transaction.id
                        }), 200

            else:
                transaction.status = 'failed'
                return jsonify({'message': payment.error}), 400

        db.session.commit()
        return jsonify({
            'message': 'Payment processed successfully',
            'transaction_id': transaction.id,
            'status': transaction.status
        }), 200

    except Exception as e:
        transaction.status = 'failed'
        db.session.commit()
        return jsonify({'message': str(e)}), 400

@app.route('/payment/execute', methods=['GET'])
def execute_payment():
    payment_id = request.args.get('paymentId')
    payer_id = request.args.get('PayerID')

    payment = paypalrestsdk.Payment.find(payment_id)

    if payment.execute({"payer_id": payer_id}):
        # تحديث حالة المعاملة
        transaction = Transaction.query.filter_by(transaction_id=payment_id).first()
        if transaction:
            transaction.status = 'completed'
            db.session.commit()

        return jsonify({'message': 'Payment executed successfully'}), 200
    else:
        return jsonify({'message': payment.error}), 400
```

```python
import pyotp

# عند تسجيل المستخدم
def setup_2fa(user):
    totp = pyotp.TOTP(pyotp.random_base32())
    user.two_factor_secret = totp.secret
    db.session.commit()
    return totp.provisioning_uri(user.email, issuer_name="SecurePay")

# عند التحقق من 2FA
def verify_2fa(user, token):
    totp = pyotp.TOTP(user.two_factor_secret)
    return totp.verify(token)
```
